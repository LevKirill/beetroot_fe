    1. function myFunc () {
        console.log('Hello World!');
      }
    2. function myFunc2 (param) {
        console.log('Привіт ' + param);
      }
    3. let myFunc3 = function (param) {
        console.log('Привіт ' + param);
      }
    4. let myFunc4 = () => {
        console.log('Hello World!');
      }
    5. let myFunc5 = () => console.log('Hello World!');
